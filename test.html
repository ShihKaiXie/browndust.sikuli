def log_add(x):
    ticks = time.time()
    txt.append(time.strftime("%Y/%m/%d %H:%M:%S", time.localtime(ticks))+'->'+unicode(x, encoding='utf-8')+'\n')
    txt.setCaretPosition(txt.getDocument().getLength());
    f.write(time.strftime("%Y/%m/%d %H:%M:%S", time.localtime(ticks))+'->'+x+'\n')
    f.flush
